# reconstructors/mle_nll_pgd.py
from .base import BaseReconstructor, ReconResult
import numpy as np

class MLENLLProjected(BaseReconstructor):
    def fit(self) -> ReconResult:
        rho = self._init_rho()  # å¯ç”¨çº¿æ€§é‡æ„ä½œä¸ºåˆå€¼
        tol = self.kwargs.get("tol", 1e-6)
        max_iter = self.kwargs.get("max_iter", 500)
        eta = self.kwargs.get("lr", 0.1)
        eps = 1e-12

        P = self.P; n = self.n
        n_sum = np.sum(n)

        def nll(rho):
            # -Î£ n_i log Tr(P_i rho)
            probs = np.array([np.trace(Pi @ rho).real for Pi in P]) + eps
            return -np.dot(n, np.log(probs))

        def grad(rho):
            # -Î£ (n_i / Tr(P_i rho)) P_i
            probs = np.array([np.trace(Pi @ rho).real for Pi in P]) + eps
            coeff = -(n / probs)
            G = sum(c * Pi for c, Pi in zip(coeff, P))
            return (G + G.conj().T) / 2  # ä¿ Hermitian

        def proj_psd_trace1(rho):
            rho = (rho + rho.conj().T) / 2
            w, V = np.linalg.eigh(rho)
            w = np.clip(w, 0.0, None)
            s = w.sum()
            if s <= eps:
                w = np.ones_like(w) / len(w)
            else:
                w = w / s
            return V @ np.diag(w) @ V.conj().T

        history = []
        f_prev = nll(rho)
        for k in range(max_iter):
            g = grad(rho)
            # ç®€æ˜“çº¿æœç´¢
            t = eta
            for _ in range(10):
                cand = proj_psd_trace1(rho - t * g)
                f_cand = nll(cand)
                if f_cand <= f_prev - 1e-4 * t * np.sum(g.conj()*g).real:
                    rho, f_prev = cand, f_cand
                    break
                t *= 0.5
            history.append(f_prev)
            if len(history) > 5 and abs(history[-5] - history[-1]) < tol:
                break

        info = {"method": "mle_nll_pgd", "nll": f_prev, "iters": len(history)}
        return ReconResult(rho, info)

    def _init_rho(self):

# reconstructors/mle_rrr.py
from .base import BaseReconstructor, ReconResult
import numpy as np

class RhoRhoRReconstructor(BaseReconstructor):
    def fit(self) -> ReconResult:
        rho = self._init_rho()
        P = self.P; n = self.n
        tau = self.kwargs.get("dilution", 0.5)
        max_iter = self.kwargs.get("max_iter", 1000)
        tol = self.kwargs.get("tol", 1e-7)
        eps = 1e-12

        def proj_psd_trace1(rho):
            rho = (rho + rho.conj().T) / 2
            w, V = np.linalg.eigh(rho)
            w = np.clip(w, 0.0, None)
            s = w.sum()
            if s < eps: w = np.ones_like(w) / len(w)
            else:       w = w / s
            return V @ np.diag(w) @ V.conj().T

        def nll(rho):
            probs = np.array([np.trace(Pi @ rho).real for Pi in P]) + eps
            return -np.dot(n, np.log(probs))

        last = nll(rho)
        for it in range(max_iter):
            probs = np.array([np.trace(Pi @ rho).real for Pi in P]) + eps
            R = sum((ni/pi) * Pi for ni, pi, Pi in zip(n, probs, P))
            A = (1 - tau) * np.eye(self.d) + tau * R
            rho = A @ rho @ A
            rho = rho / np.trace(rho).real
            rho = proj_psd_trace1(rho)

            cur = nll(rho)
            if abs(last - cur) < tol:
                break
            last = cur

        info = {"method": "mle_rrr", "nll": last, "iters": it+1, "dilution": tau}
        return ReconResult(rho, info)

    def _init_rho(self):
        return np.eye(self.d) / self.d
